# Race Riots
province_event = {
	id = 110092
	title = "EVTNAME110092"
	desc = "EVTDESC110092"
	
	trigger = {
		owner = { 
			tag = USA 
			has_country_flag = has_used_radical_reconstruction
		}
		is_overseas = no
		has_national_minority = yes
		pop_militancy = 5 #Returns true if any pop in the province has a militancy value equal to x or higher.
		NOT = { has_province_modifier = racial_tensions }
	}
	
	allow_multiple_instances = yes
	
	mean_time_to_happen = {
		months = 100 #5 years
		modifier = {
			factor = 0.5
			is_accepted_culture = no
		}
		modifier = {
			factor = 0.75
			is_core = CSA
		}
		modifier = {
			factor = 0.5
			has_province_modifier = reconstruction
		}
		modifier = {
			factor = 0.20
			controlled_by_rebels = yes
		}
		modifier = {
			factor = 0.5
			unemployment = 0.50
		}
		modifier = {
			factor = 10
			NOT = { unemployment = 0.10 }
		}
		modifier = {
			factor = 2
			crime_fighting = 0.75
		}
		modifier = {
			factor = 2
			owner = { any_owned_province = { has_province_modifier = racial_tensions } }
		}
		modifier = {
			factor = 5
			literacy = 0.8
		}
	}
	
	option = {
		name = "EVTOPTA110092" #A sad state of affair.
		owner = { prestige = -1 }
		add_province_modifier = {
			name = racial_tensions
			duration = 1825 #5 years
		}
		any_pop = { 
			limit = { 
				OR = {
					culture = dixie
					culture = afro_american
					culture = cherokee
					culture = native_american_minor
				}
			}
			militancy = 3
		}
		set_province_flag = racial_tensions
	}
}
